63  bool check(int x, int y, int p, int c, int ts)
64  {
65  if(!Ssize) return false;
66 
67  yarp::sig::ImageOf<yarp::sig::PixelInt> *active = 0;
68  if(c == 0) {
69  if(p == 0) {
70  active = &TSleftL;
71  } else if(p == 1) {
72  active = &TSleftH;
73  }
74  } else if(c == 1) {
75  if(p == 0) {
76  active = &TSrightL;
77  } else if(p == 1) {
78  active = &TSrightH;
79  }
80  }
81  if(!active) return false;
82 
83  x += Ssize;
84  y += Ssize;
85 
86  bool add = false;
87 
88  int dt = ts - (*active)(x, y);
89  if(dt < 0)
90  dt += vtsHelper::max_stamp;
91  if(dt < Tsize)
92  return add;
93 
94  (*active)(x, y) = ts;
95  for(int xi = x - Ssize; xi <= x + Ssize; xi++) {
96  for(int yi = y - Ssize; yi <= y + Ssize; yi++) {
97  int dt = ts - (*active)(xi, yi);
98  if(dt < 0) {
99  dt += vtsHelper::max_stamp;
100  (*active)(xi, yi) -= vtsHelper::max_stamp;
101  }
102  if(dt && dt < Tsize) {
103  add = true;
104  break;
105  }
106  }
107  }
108 
109  return add;
110  }
